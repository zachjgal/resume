% The headings for resume sections
\newcommand{\resheading}[1]{
\begin{tcolorbox}[colback=blue2,
                  colframe=blue1,
                  width=\textwidth,
                  arc=2mm, auto outer arc,
                 ]
  \textcolor{blue1}{\textbf{\large{#1}}}
\end{tcolorbox}
}

% Resume entry for a school experience
% <name> <location> <degree> <start date> <end date>
\newcommand{\school}[5]{
  \textbf{#1} \hfill \textit{#4} - \textit{#5}  \\
  \vspace{0.1cm}
  \textit{#3} \hfill   \text{#2}
}

% Resume entry for a previous job
% <company> <job title> <start date> <end date> <(list of strings) key skills> <location>
\newcommand{\job}[6]{
  \textbf{#1}, \text{#2} \hfill \textit{#3} - \textit{#4} \\
  \vspace{1mm}\quad\foreach \i in {#5} {\KEYSKILL{\i} } \hfill \text{#6}
}

% Resume entry for a project
% <project name> <start date> <end date> <(list of strings) key skills>
\newcommand{\project}[4]{
  \textbf{#1} \hfill \textit{#2} - \textit{#3} \\
  \foreach \i in {#4} {\KEYSKILL{\i} }
}

% Resume entry for an award. Maybe one day I'll win an award.
\newcommand{\award}[4]{
  \textbf{#1} #2 \hfill \textit{#3} #4
}
